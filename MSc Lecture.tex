\section{Finding (and using) literature efficiently}
\subsection{Google versus the rest}

\begin{frame}
\frametitle{Google versus the rest}
\begin{block}{GIYF, sometimes}
\uncover<1>{
\begin{itemize}
	\item Use \link{https://scholar.google.nl/}{Google Scholar}, not ``plain'' Google.
	\item Easiest option if you know what you're looking for.
	\item Not for books! Use \link{https://vu.on.worldcat.org/advancedsearch?databaseList=}{LibSearch}.
\end{itemize}
}
\end{block}

\begin{block}{The rest: bibliographic databases}
\uncover<2>{
\begin{itemize}
	\item More structured search
	\item Tools for expanding/limiting your search
	\item Searches in more specific (economic/business) database
\end{itemize}
}
\end{block}
\end{frame}

\subsection{Topic search}

\begin{frame}
\frametitle{Topic search}
\begin{block}{Search using keywords}
\begin{itemize}
	\item operators: `AND' or `OR' (see \link{http://images.webofknowledge.com/WOK46/help/WOS/ht_operators.html}{here} or \link{https://youtu.be/5K_yUMvt_nU}{here}),\\
	\begin{columns}
	\column{.1\linewidth}
	\column{.4\linewidth}
	\bulb $x\text{ AND }y$:
	\begin{tikzpicture}[baseline=-1mm]
		\begin{scope}[thick]
			\draw[clip] (0,0) circle[radius=3mm];
			\node at (2.5mm,0) [fill=VUBlue2, shape=circle, minimum size=6mm, inner sep=0pt] {};
		\end{scope}
		\node at (2.5mm,0) [shape=circle, minimum size=6mm, inner sep=0pt, draw=black] {};
		\node at (-1.4mm,0){$ \scriptstyle x$};
		\node at (3.9mm,0){$\scriptstyle y$};
	\end{tikzpicture}% (0.3,0) circle (.5cm);
	\column{.4\linewidth}
	\bulb $x\text{ OR }y$: \tikz[baseline=-1mm] \draw[fill=VUBlue2] (0,0) circle[radius=3mm] (2.5mm,0) circle[radius=3mm];% (0.3,0) circle (.5cm);
	\column{.1\linewidth}
	\end{columns}
	\item Also useful:\\
	\begin{tikzpicture}[anchor=west]
		\node at (1em,0) {\bulb parentheses: (\ldots)};
		\node at (1em,-3ex) {\bulb quotes: \textquotedbl\ldots\textquotedbl};
		\node at (13em,0) {\bulb wildcards: `*', `?'};
	\end{tikzpicture}
	\item (fdi OR "foreign direct investment" OR multinational) AND (tfp OR "total factor productivity")
\end{itemize}
\end{block}
\begin{block}{Bibliographic databases very powerful}
\begin{itemize}
	\item \link{http://www.ubvu.vu.nl/pub/index_oclc.cfm?SearchObjectId=7&objectid=109&ordering=1&openitem=1288}{Web of Science}: general database, widely used
	\item \link{http://www.ubvu.vu.nl/pub/index_oclc.cfm?SearchObjectId=7&objectid=109&ordering=1&openitem=182}{ABI/Inform}: business/economics, thesaurus
	\item \link{http://www.ubvu.vu.nl/pub/index_oclc.cfm?SearchObjectId=7&objectid=109&ordering=1&openitem=2055}{Business Source Elite}: business/economics, thesaurus
	\item Also use other filters: time, field of study, etc
\end{itemize}
\end{block}
\end{frame}

\end{itemize}
\end{block}

\subsection{Selecting articles}

\begin{frame}
\frametitle{How to select articles?}
\begin{block}{Rules of thumb}
\begin{itemize}
%	\item Newspaper articles, blogs etc: good for inspiration or background info, not as `serious' reference \\(but cite them if used!)
	\item Peer-reviewed work (journal articles) preferred over non-peer-reviewed (eg, working papers).
	\item (Exceptions: \link{http://www.nber.org/papers.html}{NBER}, \link{http://www.iza.org/en/webcontent/publications/papers}{IZA}, \ldots)
	\item Articles with more citations are often more influential
%	\item Ask your professor (\textit{but first do your homework}).
	\item No guarantee! Use your brain first, then your advisor.
\end{itemize}
\end{block}

\begin{block}{For your paper (if you need $X$ articles)}
\begin{itemize}
	\item Do topic search $\Rightarrow$ \textit{many} articles
	\item Sort by citations
	\item Select approx $1.5\times X$ top articles
	\item Scan / read superficially (abstract/intro/results)
	\item Discard what isn't useful
	\item Repeat \& adjust search if necessary
\end{itemize}
\end{block}
\end{frame}

\subsection{Snowball search}

\againframe<2>{workflow}

\begin{frame}
\frametitle{How to search: Forward and backward snowball}
\begin{block}{Backward snowball}
\begin{itemize}
	\item Which articles were used for this research?
	\item Check the references in an article
\end{itemize}
\end{block}

\begin{block}{Forward snowball}
\begin{itemize}
	\item How was this article used in other research?
	\item Google Scholar: ``Cited by: \ldots''
    \item WoS: ``Times cited: \ldots''
\end{itemize}
\end{block}
\end{frame}

\begin{block}{For your paper}
\begin{itemize}
	\item Select 5--10 of most interesting papers
	\item Backward snowball to see if you missed anything
	\item Forward snowball: \\
	Do any recent papers cite many of your selected papers?
	\item If yes: scan for interest. \\
	New theories, new data, new controversy?
\end{itemize}
\end{block}

\subsection{Proper references}

\begin{frame}
\frametitle{Leave your finger prints}
\begin{block}{Proper references}
\begin{itemize}
	\item Plagiarism is \alert{serious} academic crime!
	\item Most common in Econ \& Finance: author-year format
	\item author(s) and year (in parenthesis) of publication in text
	\item Full reference in bibliography\\(At least: authors, year of pub, full article title, journal title, journal volume/issue, pages.  Include a \link{https://en.wikipedia.org/wiki/Digital_object_identifier}{doi} if you can.)
	\item See \link{http://libguides.vu.nl/B_business_admin/incorporating_literature}{here} for more
\end{itemize}
\end{block}

\begin{block}{Invest some time in (semi-)automation}
\begin{itemize}
	\item \link{http://endnote.com/}{Endnote}: supported by \link{http://www.ub.vu.nl/en/help-advice/tools/endnote/index.aspx}{library}.
	\item \link{https://www.mendeley.com/}{Mendeley}, \link{https://www.zotero.org/}{Zotero}: online \& free
	\item \link{http://www.jabref.org/}{JabRef}, \link{http://bibdesk.sourceforge.net/}{BibDesk}: \LaTeX{} users
	\item Ready-made formats, integration with word processors.
\end{itemize}
\end{block}
\end{frame}
